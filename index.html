// 分页获取视频示例 async function getVideosPage(page = 1, limit = 10) { const response = await fetch('/videos.json'); const data = await response.json(); const startIndex = (page - 1) * limit; const endIndex = startIndex + limit; const paginatedVideos = data.videos.slice(startIndex, endIndex); return { videos: paginatedVideos, pagination: { page, limit, total: data.videos.length, totalPages: Math.ceil(data.videos.length / limit) } }; }
